Tab2:AddButton({
    Name = "• 15 JJ Tool",
    Callback = function()
        local player = game.Players.LocalPlayer
        local backpack = player:FindFirstChild("Backpack")
        local character = player.Character or player.CharacterAdded:Wait()

        if not backpack or not character then
            warn("Backpack veya Character bulunamadı!")
            return
        end

        local toolName = "15 JJ ÇEK!"

        -- Eğer tool zaten varsa, tekrar oluşturma
        for _, item in ipairs(backpack:GetChildren()) do
            if item:IsA("Tool") and item.Name == toolName then
                print("Tool zaten var, tekrar eklenmeyecek.")
                return
            end
        end

        -- Yeni Tool oluştur
        local tool = Instance.new("Tool")
        tool.Name = toolName
        tool.RequiresHandle = false
        tool.Parent = backpack

        -- Mesaj Listesi
        local messages = {
            "BİR", "İKİ", "ÜÇ", "DÖRT", "BEŞ",
            "ALTI", "YEDİ", "SEKİZ", "DOKUZ", "ON",
            "ON BİR", "ON İKİ", "ON ÜÇ", "ON DÖRT", "ON BEŞ"
        }

        local interval = 3 -- Mesajlar arası toplam süre (saniye)
        local isActive = false -- Tekrar kullanım engelleme

        -- **Zıplama işlemini doğrudan karaktere bağlayan fonksiyon**
        local function forceJump()
            local humanoid = character:FindFirstChildOfClass("Humanoid")
            if humanoid then
                humanoid:ChangeState(Enum.HumanoidStateType.Jumping) -- **Kesin zıplatır**
            else
                warn("Humanoid bulunamadı!")
            end
        end

        -- **Mesaj ve Zıplama Fonksiyonu**
        local function sendMessages()
            if isActive then return end
            isActive = true

            local chatService = game:GetService("TextChatService")
            local channel = chatService.TextChannels.RBXGeneral

            if channel then
                for _, message in ipairs(messages) do
                    channel:SendAsync(message) -- Mesaj gönder
                    task.wait(0.5) -- 0.5 saniye bekle
                    forceJump() -- **Yeni zıplama fonksiyonunu çağır**
                    task.wait(interval - 0.5) -- Kalan süreyi bekle
                end
            else
                warn("Chat kanalı bulunamadı!")
            end

            isActive = false
        end

        -- **Tool Kullanıldığında Mesaj Gönderme**
        tool.Activated:Connect(sendMessages)
    end
})

-- Infinite Yield admin butonunu ekliyoruz
Tab5:AddButton({
    Name = "• İnfinite Yield Admin",
    Callback = function()
        -- İnfinite Yield scriptini CodeX ile çalıştırmak için
        local scriptUrl = "https://raw.githubusercontent.com/EdgeIY/infiniteyield/master/source"
        
        -- HttpGet komutu ile scripti almak
        local success, response = pcall(function()
            return game:HttpGet(scriptUrl)
        end)
        
        if success and response then
            -- loadstring ile scripti çalıştırıyoruz
            loadstring(response)()
        else
            -- Hata durumunda kullanıcıya uyarı veriyoruz
            warn("Script yüklenemedi!")
        end
    end
})

Tab5:AddButton({
    Name = "• Infinite Jump",
    Callback = function()
        local InfiniteJumpEnabled = true
        game:GetService("UserInputService").JumpRequest:Connect(function()
            if InfiniteJumpEnabled then
                local humanoid = game:GetService("Players").LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
                if humanoid then
                    humanoid:ChangeState("Jumping")
                end
            end
        end)
    end
})

Tab5:AddButton({
    Name = "• FPS Booster (Potato Graphics)",
    Callback = function()
        -- Grafik düşürme fonksiyonu
        local function optimizeGraphics()
            -- Gereksiz parçacık efektlerini kaldır
            for _, obj in pairs(game:GetDescendants()) do
                if obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Smoke") or obj:IsA("Fire") then
                    obj:Destroy()
                elseif obj:IsA("BasePart") then
                    obj.Material = Enum.Material.SmoothPlastic
                    obj.CastShadow = false
                    obj.Reflectance = 0 -- Yansımaları kapat
                elseif obj:IsA("Decal") then
                    obj:Destroy()
                end
            end

            -- Aydınlatma ayarlarını optimize et
            local lighting = game:GetService("Lighting")
            lighting.GlobalShadows = false
            lighting.FogEnd = 1e9 -- Sonsuz görüş alanı
            lighting.Brightness = 1 -- Minimum parlaklık
            lighting.OutdoorAmbient = Color3.fromRGB(128, 128, 128) -- Nötr ortam ışığı
        end

        -- Optimizasyonu başlat
        optimizeGraphics()

        -- Yeni eklenen nesneleri optimize et
        game.DescendantAdded:Connect(function(obj)
            if obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Smoke") or obj:IsA("Fire") then
                obj:Destroy()
            elseif obj:IsA("BasePart") then
                obj.Material = Enum.Material.SmoothPlastic
                obj.CastShadow = false
                obj.Reflectance = 0
            elseif obj:IsA("Decal") then
                obj:Destroy()
            end
        end)

        -- Bildirim gönder
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "FPS Booster",
            Text = "Grafikler başarıyla düşürüldü.",
            Duration = 5
        })
    end
})
