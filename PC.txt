Tab5:AddButton({
    Name = "• FPS Booster (Potato Graphics)",
    Callback = function()
        -- Grafik düşürme fonksiyonu
        local function optimizeGraphics()
            -- Gereksiz parçacık efektlerini kaldır
            for _, obj in pairs(game:GetDescendants()) do
                if obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Smoke") or obj:IsA("Fire") then
                    obj:Destroy()
                elseif obj:IsA("BasePart") then
                    obj.Material = Enum.Material.SmoothPlastic
                    obj.CastShadow = false
                    obj.Reflectance = 0 -- Yansımaları kapat
                elseif obj:IsA("Decal") then
                    obj:Destroy()
                end
            end

            -- Aydınlatma ayarlarını optimize et
            local lighting = game:GetService("Lighting")
            lighting.GlobalShadows = false
            lighting.FogEnd = 1e9 -- Sonsuz görüş alanı
            lighting.Brightness = 1 -- Minimum parlaklık
            lighting.OutdoorAmbient = Color3.fromRGB(128, 128, 128) -- Nötr ortam ışığı
        end

        -- Optimizasyonu başlat
        optimizeGraphics()

        -- Yeni eklenen nesneleri optimize et
        game.DescendantAdded:Connect(function(obj)
            if obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Smoke") or obj:IsA("Fire") then
                obj:Destroy()
            elseif obj:IsA("BasePart") then
                obj.Material = Enum.Material.SmoothPlastic
                obj.CastShadow = false
                obj.Reflectance = 0
            elseif obj:IsA("Decal") then
                obj:Destroy()
            end
        end)

        -- Bildirim gönder
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "FPS Booster",
            Text = "Grafikler başarıyla düşürüldü.",
            Duration = 5
        })
    end
})
